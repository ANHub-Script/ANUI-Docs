Usage Examples

Learn how to use ANUI components in your scripts

Creating a Basic Window

Create your first ANUI window with customizable properties

Luau 
-- Load the library
local ANUI = loadstring(game:HttpGet(
    "https://raw.githubusercontent.com/ANHub-Script/ANUI/refs/heads/main/dist/main.lua"
))()

-- Create a window
local Window = ANUI:CreateWindow({
    Title = "My First Script",
    Author = "by YourName",
    Folder = "myscript",
    Icon = "rbxassetid://84366761557806",
    IconSize = 44,
    NewElements = true,
    HideSearchBar = false,
    OpenButton = {
        Title = "My UI",
        CornerRadius = UDim.new(1, 0),
        StrokeThickness = 3,
        Enabled = true,
        Draggable = true,
        OnlyMobile = false,
        Color = ColorSequence.new(
            Color3.fromHex("#30FF6A"), 
            Color3.fromHex("#e7ff2f")
        )
    }
})

-- Add a version tag
Window:Tag({
    Title = "v1.0.0",
    Icon = "github",
    Color = Color3.fromHex("#1c1c1c")
})

Using Toggles and Checkboxes

Create interactive toggles with animations and icons

Luau 
-- Basic Toggle
local MyToggle = Tab:Toggle({
    Title = "Enable Feature",
    Desc = "Turn this feature on or off",
    Icon = "settings",
    IconSize = 15,
    Value = true, -- Default value
    Callback = function(value)
        print("Toggle is now:", value)
    end
})

-- Checkbox Variant
Tab:Toggle({
    Title = "Agree to Terms",
    Type = "Checkbox",
    Value = false,
    Callback = function(value)
        print("Checkbox:", value)
    end
})

-- Programmatic Control
MyToggle:Set(false, true, false) -- Set to false, trigger callback, with animation
MyToggle:SetTitle("Feature Disabled")
MyToggle:SetDesc("Click to enable")

Creating Buttons

Add clickable buttons with custom styling

Luau 
-- Basic Button
Tab:Button({
    Title = "Click Me",
    Desc = "This is a button",
    Icon = "mouse-pointer-click",
    IconAlign = "Left",
    Justify = "Center",
    Callback = function()
        print("Button clicked!")
    end
})

-- Colored Button
Tab:Button({
    Title = "Execute Script",
    Color = Color3.fromHex("#305dff"),
    Icon = "play",
    Callback = function()
        -- Your code here
    end
})

-- Locked Button
Tab:Button({
    Title = "Premium Feature",
    Desc = "Unlock to access",
    Locked = true,
    Icon = "lock"
})

Working with Dropdowns

Create selection menus with search and icons

Luau 
-- Simple Dropdown
Tab:Dropdown({
    Title = "Select Option",
    Values = {"Option 1", "Option 2", "Option 3"},
    Value = "Option 1",
    SearchBarEnabled = true,
    Callback = function(value)
        print("Selected:", value)
    end
})

-- Dropdown with Icons
Tab:Dropdown({
    Title = "Choose Weapon",
    Values = {
        {Title = "Sword", Icon = "sword"},
        {Title = "Bow", Icon = "target"},
        {Title = "Staff", Icon = "wand"}
    },
    Value = {Title = "Sword", Icon = "sword"},
    SearchBarEnabled = true,
    Callback = function(option)
        print("Selected:", option.Title)
    end
})

-- Multi-select Dropdown
Tab:Dropdown({
    Title = "Select Multiple",
    Values = {"A", "B", "C", "D"},
    Multi = true,
    Value = {"A", "B"},
    Callback = function(values)
        print("Selected items:", table.concat(values, ", "))
    end
})

Using Sliders

Create value sliders with custom ranges

Luau 
-- Integer Slider
local WalkSpeedSlider = Tab:Slider({
    Title = "Walk Speed",
    Desc = "Adjust your walking speed",
    Value = {
        Min = 16,
        Max = 100,
        Default = 16
    },
    Callback = function(value)
        game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = value
    end
})

-- Decimal Slider
Tab:Slider({
    Title = "Brightness",
    Step = 0.1,
    Value = {
        Min = 0,
        Max = 1,
        Default = 0.5
    },
    Callback = function(value)
        game.Lighting.Brightness = value
    end
})

-- Programmatic Control
WalkSpeedSlider:Set(50)
